\begin{frame}\frametitle{Convective Derivative: Definition}
		%
\begin{equation}
	\paren{\tdt{f(\bx+ \mathbf{u}\bdt, t+\bdt)}}_{\text{stream }\mathbf{u}} 
		= \lim\limits_{\bdt \to 0} \frac{f(\rd{\bx + \bu \bdt}, t + \bdt) - \fxt} {\bdt}

\end{equation}
		%
\end{frame}

\begin{frame}\frametitle{Convective Derivative: Resolution}
\center
\mg{Method: Taylor Approximation Theorem}\\
		%
\begin{equation}
\begin{split}
	f(\bx+ \mathbf{u}\bdt, t+\bdt) 
		&= f\paren{x_{1}, t + \bdt} + u_{1}\bdt \fxtd{1}{\tbdt} \\
		&+ f\paren{x_{2}, t + \bdt} + u_{2}\bdt \fxtd{2}{\tbdt} \\
		&+ f\paren{x_{3}, t + \bdt} + u_{3}\bdt \fxtd{3}{\tbdt} + \mathcal{O}\paren{\bdt^{2}}\\[15pt]
		%\pause
		&= f\paren{\bx, t + \bdt} + \bdt\, \bu \cdot \nabla f\paren{\bx, t + \bdt} + \mathcal{O}\paren{\bdt^{2}}
\end{split}
\label{eq:convective-definition-resolution}
\end{equation}
		%
\end{frame}
%
\begin{frame}\frametitle{Convective Derivative: Resolution}
\center
%Method: Taylor Approximation Theorem\\
		%
\begin{equation}
\begin{split}
	\lim\limits_{\bdt \to 0} \frac{ f(\rd{\bx + \bu \bdt}, t + \bdt) } { \bdt }
		&= \lim\limits_{\bdt \to 0} \frac{ f\paren{\bx, t + \bdt} } { \bdt } + \cancel{\bdt} \frac{ \rd{\bu \cdot \nabla} f\paren{\bx, t + \bdt} } { \cancel{\bdt} } \\[10pt]
		&= \lim\limits_{\bdt \to 0} \frac{ f\paren{\bx, t + \bdt} } { \bdt } + \rd{\bu \cdot \nabla} f\paren{\bx,t}
\end{split}
\label{eq:convective-definition-resolution-term}
\end{equation}
		%
\end{frame}

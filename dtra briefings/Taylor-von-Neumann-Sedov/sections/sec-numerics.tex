\subsection{Development of Sedov Solution}
%
\begin{frame}\frametitle{\href{https://en.wikipedia.org/wiki/Compressible\_flow}{Compressible Hydrodynamics}: Equations}
% sedum p. 148, IV-1.3
\begin{equation}
\begin{array}{rcl}
	\pdt{v} + v \pd{v}{r} + \frac{1}{\rho}\pd{t}{r} &=& 0, \\[10pt]
	\pdt{\rho} + \pd{\rho v}{r} + (v-1) \frac{\rho v}{r} &=& 0, \\[10pt]
	\pdt{}\sfrac + v \pd{}{r} \sfrac &=& 0.
\label{eq:s1.3}
\end{array}
\end{equation}
\end{frame}
%
\begin{frame}\frametitle{\href{https://en.wikipedia.org/wiki/Compressible\_flow}{Compressible Hydrodynamics}: Equations}
\begin{align}
	\rho_{t} + v \rho_{r} \frac{\rho}{r^{j-1}} \paren{r^{j-1} v}_{r} &= 0 \\
	v_{t} + v v_{r} + \frac{1}{\rho} p_{r} &= 0 \\
	\paren{ \frac{p}{\rho^{\gamma}} }_{t} + u \paren{ \frac{p}{\rho^{\gamma}} }_{r} &= 0 \\
	\paren{ \gamma - 1} e &= \frac{p}{\rho}
\end{align}
\end{frame}

\begin{frame}\frametitle{Post shock state}
\center
\begin{equation}
\begin{array}{rl}
	v_{2} &= \displaystyle\frac{2}{\gamma + 1} U \\[15pt]
	\rho_{2} &= \displaystyle\frac{\gamma + 1}{\gamma - 1}\rho_{1} \\[15pt]
	p_{2} &= \displaystyle\frac{2}{\gamma + 1} \rho_{1} U^{2}
\end{array}
\label{eq:post-shock}
\end{equation}
\end{frame}
%
\begin{frame}\frametitle{$V^{*}$, $V_{2}$}
\center
Post-shock state:
\begin{equation}
	V_{2} = \frac{4}{\paren{j+2-\omega}{\gamma+1}}	
\label{eq:v2}
\end{equation}
\ \\[20pt]
Singular point:
\begin{equation}
	V^{*} = \frac{2}{\paren{ \gamma - 1 }j+2}	
\label{eq:vstar}
\end{equation}
\end{frame}

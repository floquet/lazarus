\subsection{Intercept Only}

\begin{frame}\frametitle{Model of Heat Flow}
\center
Temperature as a \bl{linear function} of position
	%
\begin{equation}
		%
	y(x) = \bazero + \baone x
		%
\label{eq:model}
\end{equation}
%
Solution parameters: \\
$\bazero$ intercept \\
$\baone$ slope
	%
\end{frame}

\begin{frame}\frametitle{Least Squares Inputs and Outputs}
	%
Inputs: \\
%
\begin{enumerate}
	\item Model function, $y(x) = \bazero + \baone x$
	\item Sequence of measurements, $\seqinputs{x}{y}$
\end{enumerate}
	\ \\[10pt]
	%
	\pause
	%
Outputs: Model parameters\\
\begin{enumerate}
	\item $\bazero \pm \bl{\sigma_{0}}$, intercept with error
	\item $\baone \pm \bl{\sigma_{1}}$, slope with error
\end{enumerate}
\end{frame}
%
\begin{frame}\frametitle{Linear System}
\begin{equation}
	\begin{array}{ccccc cccc}
			%
		\A{} & a & = & y \\
			%
		\mat{cc}{\one & x} & \littlea & = & \mat{c}{y}\\
			%
		\pause
			%
		\bevA{} & \littlea & = & \bevy 
			%
	\end{array}
\label{eq:bevington normal equations problem and solution}
\end{equation}
\end{frame}
%
%
\begin{frame}\frametitle{Linear System: Dramatis Personae}
\begin{equation}
	\begin{array}{cccc}
			%
		\A{} & a & = & y \\
			%
		\mat{cc}{\one & x} & \littlea & = & \mat{c}{y}\\
			%
	\end{array}
\label{eq:Dramatis Personae}
\end{equation}
\begin{enumerate}
	\item $\A{}\in\real{\by{m}{2}}_{2}$: design matrix -- \mg{mesh}
	\item $y\in\real{m}$: data vector -- \mg{measurements}
	\item $a\in\real{m}$: solution vector
\end{enumerate}
\end{frame}
%
\begin{frame}\frametitle{Notation: Inner Product}
Given $x, y \in \cmplx{m}$
	%
\begin{equation}
\begin{split}
	\inner{x, y} = x^{*}y = x \cdot y 
		&= \sum \bar{x}_{k}y_{k} \\
		&= \int_{\Omega} \bar{x} y d\Omega
\end{split}
\end{equation}
\end{frame}
%
\begin{frame}\frametitle{System of Linear Equations}
Model Function: \\
\begin{equation}
	y(x) = \bazero + \baone x
%\label{eq:}
\end{equation}
\ \\
A sequence of $m$ measurements:\\
\begin{equation}
\begin{array}{ccc ccc}
	\bazero &+& \baone & x_{1} &=& y_{1} \\
		 &&& \vdots && \vdots \\
	\bazero &+& \baone & x_{m} &=& y_{m} \\
\end{array}
\label{eq:system}
\end{equation}
\end{frame}

\begin{frame}\frametitle{What Are We Minimizing?}
\rd{Residual error} vector:
\begin{equation}
		%
	\rd{r(\bk{a})} = \A{}a - y
		%
%\label{eq:}
\end{equation}
	%
	\pause
	%
Minimize length of \rd{residual error} vector:
\begin{equation}
\begin{split}
		%
	M(a) &= \innerecho{\rd{r(\bk{a})}} \\
		%
		& = \normt{\A{}a - y} \\
		%
		& = \normt{a_{0} \one + a_{1} x - y} \\
		%
\end{split}
%\label{eq:}
\end{equation}
	%
	\pause
	%
\mg{Archaic (scalar) notation:
\begin{equation*}
		%
	M(a) = \sum_{k=1}^{m} \paren{a_{0} + a_{1} x_{k} - y_{k}}^{2}
		%
%\label{eq:}
\end{equation*}}
\end{frame}
%

\subsection{Nonlinear Formulation}
%
\begin{frame}\frametitle{From the Beginning...}
	{\huge{
	\begin{equation}
		(\bl{a}, \bl{b})_{LS}\equiv \argmin\limits_{(\bl{a},\bl{b})\in\mathbb{R}^{2}} \normt{\bl{a} t^{\bl{b}} - R}
	\tag{\ref{eq:solution definition}}
	\end{equation}
	}}
	\ \\
	$b$ is \bl{nonlinear}, but $a$ is \bl{linear}:\\
	no need for \href{https://en.wikipedia.org/wiki/Levenberg\%E2\%80\%93Marquardt_algorithm}{Levenberg-Marquardt}
\end{frame}

\begin{frame}\frametitle{Minimization Problem}
Residual error function: \\
\begin{equation}
\begin{split}
	\tilde{\phi}(a,b) &= \innerecho{a t^{b}-R} \\
		&= a^{2} \innerecho{t^{b}} + \innerecho{R} - 2a \inner{t^{b},R}
\end{split}
\label{eq:phi-hat}
\end{equation}
\ \\
Solve
\begin{equation}
\nabla \tilde{\phi} = \mathbf{0}
%\label{eq:}
\end{equation}
\end{frame}
%
\begin{frame}\frametitle{Dimensional Reduction}
\begin{equation}
\begin{split}
	\partial_{a} \tilde{\phi}(a,b) &=	0 \\
	2 a \innerecho{t^{b}} - 2\inner{t^{b},R} &=	0
\end{split}
\label{eq:partiala}
\end{equation}
\ \\[20pt]
\pause
\begin{equation}
	a_{*} = \frac{ \inner{t^{b},R} } { \innerecho{t^{b}} }
\label{eq:aopt}
\end{equation}
\end{frame}

\subsection{Least Squares Exemplar Problem}
%
\begin{frame}\frametitle{\lsq \ Exemplar Problem}
	%
\begin{equation}
		%
	\exemplartall \littlea = \mat{c}{y_{1} \\ y_{2} \\ y_{3}}
		%
\label{eq:exemplar}
\end{equation}
	%
\end{frame}
%
\begin{frame}\frametitle{\lsq \ Exemplar Problem}
	%
Problem:
\begin{equation}
		%
	\mat{cc}{1 & 0 \\ 0 & 1 \\ \rzero & \rzero} \littlea = \mat{c}{y_{1} \\ y_{2} \\ \rd{y_{3}}}
		%
\tag{\ref{eq:exemplar}}
\end{equation}
	%
\end{frame}
%
\begin{frame}\frametitle{\lsq \ Exemplar Solution}
	%
Problem:
\begin{equation}
		%
	\mat{cc}{1 & 0 \\ 0 & 1 \\ \rzero & \rzero} \littlea = \mat{c}{y_{1} \\ y_{2} \\ \rd{y_{3}}}
		%
\tag{\ref{eq:exemplar}}
\end{equation}
	%
Solution:
	%
\begin{equation}
		%
	\bl{\littlea} = \bl{\mat{c}{y_{1} \\ y_{2}}}
		%
\label{eq:solution}
\end{equation}
\end{frame}

\begin{frame}\frametitle{\lsq \ Exemplar Error}
	%
Problem:
\begin{equation}
		%
	\mat{cc}{1 & 0 \\ 0 & 1 \\ \rzero & \rzero} \littlea = \mat{c}{y_{1} \\ y_{2} \\ 0}
		%
\tag{\ref{eq:exemplar}}
\end{equation}
	%
Solution:
	%
\begin{equation}
		%
	\bl{\littlea} = \bl{\mat{c}{y_{1} \\ y_{2}}}
		%
\tag{\ref{eq:solution}}
\end{equation}
%
Residual error:
	%
\begin{equation}
		%
	\rd{r^{2}} = \normts{\bl{\mat{c}{y_{1} \\ y_{2} \\ 0}} - \mat{c}{y_{1} \\ y_{2} \\ \rd{y_{3}}}} = \abs{y_{3}}^{2}
		%
\label{eq:error}
\end{equation}
%
\end{frame}

\subsection{Least Squares Problem and Solution}

\begin{frame}\frametitle{Problem Statement}
Consider the linear equation
$$\axeb$$
\begin{enumerate}
	\item $\aicmnr$: design matrix
	\item $b\in\cmplxm$: data vector
	\item $\xicn$: solution vector
\end{enumerate}
\end{frame}
%
\begin{frame}\frametitle{\lsq \ Solution}
The \bl{\lsq \ Solution} is defined as
$$\xlsdefault$$
\end{frame}


\begin{frame}\frametitle{\lsq \ Solution}
The \bl{\lsq \ Solution} is expressed as
\begin{equation}
		\begin{array}{cccccl}
			\xls 
			%& = & \xblue{}} & + & \xred \\
			& = & \bsolnlsbl{b}{y}  \\
			& = & \Apb & + & \pna{}y \\%\myline
			& = & \xp & + & \xh \\
%			& = & \bl{x_{\mathcal{R}}} \\
%			& = & \bl{x_{\scriptstyle\rnga{*}}} \\
			& = & \xblue{} & + & \xred
		\end{array}
\label{eq:definition:general lsq solution}
\end{equation}
\end{frame}

\begin{frame}\frametitle{Dramatis Personae}
%
\lsq \ Residual Error Vector
%
\begin{equation}
	\rd{r(x)} = \Ax - b
\end{equation}
%
\lsq \ Merit Function
%
\begin{equation}
		%
	M(x) = \innerecho{\rd{r(\bk{x})}} = \innerecho{\axmb}
		%
%\label{eq:}
\end{equation}
%
Least Total Squared Error
%
\begin{equation}
		%
	t^{2} = \innerecho{\rd{r(\bl{\xls})}}
		%
%\label{eq:}
\end{equation}
%
\lsq \ Uncertainty \mg{(Standard Deviation)}
%
\begin{equation}
		%
	\sigma^{2}_{k} =  \frac{t^{2}} {m-n} \wxi{*}_{k,k}, \quad \kto{n}
		%
%\label{eq:}
\end{equation}
%
\end{frame}

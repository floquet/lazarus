\subsection{Taylor's Results}
%
\begin{frame}\frametitle{Computing \yl{Yield}}
Similarity Argument (Dimensional Analysis):\\
\begin{equation}
		%
	R = S(\gamma) \sqrt[5]{\frac{\rd{E} t^{2}}{\rho_{0}} }
		%
\label{eq:taylor}
\end{equation}
\pause
Finding, yield $E$:
\begin{enumerate}
	\item $R$: measured (photo)
	\item $t$: measured (photo)
	\item $\rho_{0} = 0.00123$ g/c$^{3}$: (given)
	\pause
	\item $\rd{S(\gamma)}$: Taylor's second masterpiece
\end{enumerate}
\end{frame}

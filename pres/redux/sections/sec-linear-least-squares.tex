\begin{frame}\frametitle{Least Squares Problem}
	\huge{
	\begin{equation}
		(\bl{a}, \bl{b})_{LS}\equiv \argmin\limits_{(\bl{a},\bl{b})\in\mathbb{R}^{2}} \normt{\bl{a} t^{\bl{b}} - R}
	\label{eq:solution definition}
	\end{equation}
	}
\end{frame}

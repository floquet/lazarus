\subsubsection{The Normal Equations in Theory}
\begin{frame}\frametitle{The Normal Equations in Theory}
	%
	$$\AsA x =  \A{*}b$$
	%
\begin{enumerate}
	\item Always consistent
	\item For $\rho = n$, solution is \bl{unique}
	\item For $\rho = n$, solution is $\bl{\xls} = \Anormal b$
\end{enumerate}
	%
\end{frame}
%
